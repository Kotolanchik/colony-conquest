### 2.4 Алгоритм Генерации Персонажа

```csharp
public struct CharacterGenerator
{
    public static SettlerIdentity GenerateIdentity(GenerationParams params)
    {
        // 1. Определение пола и возраста
        var gender = Random.value < 0.5f ? 0 : 1;
        var age = Random.Range(16, 81);
        
        // 2. Генерация имени из пула фракции
        var name = NameDatabase.GetRandom(params.Faction, gender);
        
        // 3. Генерация seed для внешности
        var appearanceSeed = (uint)Random.Range(0, int.MaxValue);
        
        return new SettlerIdentity { ... };
    }
    
    public static PersonalityTraits GenerateTraits(GenerationParams params)
    {
        var traits = new PersonalityTraits();
        var incompatibleMask = 0u;
        
        // Базовое количество черт: 2-4
        int traitCount = Random.Range(2, 5);
        
        for (int i = 0; i < traitCount && i < 8; i++)
        {
            // Выбор черты с учётом весов и несовместимости
            byte traitId = SelectWeightedTrait(incompatibleMask);
            
            // Интенсивность: обычно -1 до +2
            sbyte intensity = (sbyte)Random.Range(-1, 3);
            
            traits.ActiveTraits.Add(new TraitSlot {
                TraitId = traitId,
                Intensity = intensity,
                Source = 0 // Врождённая
            });
            
            // Обновление маски несовместимости
            incompatibleMask |= TraitDatabase.GetIncompatibilityMask(traitId);
        }
        
        return traits;
    }
    
    public static SkillSet GenerateSkills(SettlerIdentity identity, PersonalityTraits traits)
    {
        var skills = new SkillSet();
        
        // Все навыки начинаются с уровня 0
        foreach (var skillId in Enum.GetValues(typeof(SkillId)))
        {
            var skill = new Skill {
                SkillId = (byte)skillId,
                Level = 0,
                Experience = 0,
                ExperienceToNext = CalculateXPForLevel(1),
                PassionLevel = GeneratePassion(traits, (byte)skillId),
                NaturalCap = GenerateNaturalCap(identity),
                LearnedCap = 20,
                TotalUses = 0
            };
            skills.Skills.Add(skill);
        }
        
        // Бонусные очки для стартовых навыков (возраст * 2)
        int bonusPoints = identity.Age * 2;
        DistributeBonusPoints(ref skills, bonusPoints, traits);
        
        return skills;
    }
    
    public static MentalConditions GenerateMentalConditions(
        PersonalityTraits traits, 
        GenerationParams params)
    {
        var conditions = new MentalConditions();
        
        // Шанс фобии: 15% базовый + модификаторы от черт
        float phobiaChance = 0.15f;
        if (HasTrait(traits, 1)) phobiaChance += 0.2f; // Трусливый
        if (HasTrait(traits, 15)) phobiaChance += 0.1f; // Умственно отсталый
        
        if (Random.value < phobiaChance)
        {
            conditions.Phobias.Add(new Phobia {

### 3.2 Формулы Роста Навыков

```
// Базовый опыт за действие
BASE_XP = ActionDifficulty * 10 * QualityMultiplier

// Множитель страсти
PASSION_MULTIPLIER = 1.0 + (PassionLevel * 0.5)
// Passion 0 = 1.0x, Passion 1 = 1.5x, Passion 2 = 2.0x, Passion 3 = 2.5x

// Множитель таланта
APTITUDE_MULTIPLIER = 1.0 + (AptitudeLevel * 0.2)
// Aptitude 1 = 1.2x, Aptitude 5 = 2.0x

// Множитель обучения (обратно пропорционален уровню)
LEARNING_MULTIPLIER = 2.0 / (1 + Level * 0.1)

// Итоговый опыт
FINAL_XP = BASE_XP * PASSION_MULTIPLIER * APTITUDE_MULTIPLIER * LEARNING_MULTIPLIER

// Потолок таланта
if (Level >= NaturalCap && Random.value > 0.1f)
    XP_GAIN *= 0.1f;  // 90% шанс потери XP выше потолка

// Проверка декрементирования
if (DaysSinceLastUse > 30 && Level > 0)
{
    float decayChance = (DaysSinceLastUse - 30) * 0.01f;
    if (Random.value < decayChance)
    {
        Experience -= (int)(MaxExperience * 0.05f);
        if (Experience < 0) LevelDown();
    }
}
```

### 3.3 Потолки Таланта

```
NATURAL_CAP генерация:
- Случайное значение 8-18
- +2 если Гений
- -2 если Умственно отсталый
- +1 за каждый талант в категории

LEARNED_CAP:
- Начинается равным NaturalCap
- Может быть увеличен через:
  * Обучение от мастера (+1 за 30 дней)
  * Уникальные события (+1-2)
  * Импланты/улучшения (+1-3)
  * Максимум: 20

ЭФФЕКТ ПРЕВЫШЕНИЯ ПОТОЛКА:
- XP выше потолка: 90% теряется
- Шанс "прорыва": 1% за каждые 100 XP "впустую"
- При прорыве: NaturalCap += 1 (редкое событие)
```

### 3.4 Примеры Расчёта

```

- Талант: 1 (+20%)

PASSION_MULT = 1 + (2 * 0.5) = 2.0
APTITUDE_MULT = 1.2
LEARNING_MULT = 2.0 / (1 + 3 * 0.1) = 1.54

FINAL_XP = 20 * 2.0 * 1.2 * 1.54 = 73.6 ≈ 74 XP

public class BatchProcessingSystem : SystemBase
{
    private const int BATCH_SIZE = 1000;
    
    protected override void OnUpdate()
    {
        var query = GetEntityQuery(
            ComponentType.ReadWrite<NeedsState>(),
            ComponentType.ReadOnly<PhysiologyState>()
        );
        
        var chunks = query.CreateArchetypeChunkArray(Allocator.TempJob);
        
        // Обработка чанками
        for (int i = 0; i < chunks.Length; i += BATCH_SIZE)
        {
            int batchEnd = math.min(i + BATCH_SIZE, chunks.Length);
            ProcessBatch(chunks, i, batchEnd);
        }
        
        chunks.Dispose();
    }
}

        var needsArray = chunk.GetNativeArray(NeedsType);
        var physioArray = chunk.GetNativeArray(PhysiologyType);
        
        for (int i = 0; i < chunk.Count; i++)
        {
            var needs = needsArray[i];
            var physio = physioArray[i];
            
            // Обновление голода
            needs.Hunger += 0.5f * DeltaTime;
            if (physio.Health < 50) needs.Hunger += 0.3f * DeltaTime;
            
            // Обновление жажды
            needs.Thirst += 1.0f * DeltaTime;
            
            // Обновление усталости
            needs.Rest += 0.3f * DeltaTime;
            
            needsArray[i] = needs;
        }
    }
}

// Сохранение состояния поселенцев
public class SettlerSaveSystem : SystemBase
{
    public void SaveState(string savePath)
    {
        var saveData = new SettlerSaveData();
        
        Entities
            .WithAll<SettlerIdentity>()
            .ForEach((Entity entity, 
                in SettlerIdentity identity,
                in SkillSet skills,
                in PsychologyState psych,
                in PhysiologyState physio,
                in SocialBonds bonds) =>
            {
                var settlerData = new SettlerData
                {
                    Identity = identity,
                    Skills = skills,
                    Psychology = psych,
                    Physiology = physio,
                    Bonds = bonds.ToArray(),
                    Stats = GetComponent<SettlerStats>(entity)
                };
                
                saveData.Settlers.Add(settlerData);
            }).Run();
        
        // Сериализация в JSON/Binary
        File.WriteAllBytes(savePath, Serialize(saveData));
    }
    
    public void LoadState(string savePath)
    {
        var saveData = Deserialize<SettlerSaveData>(
            File.ReadAllBytes(savePath));
        
        foreach (var settler in saveData.Settlers)
        {
            var entity = EntityManager.CreateEntity(
                typeof(SettlerIdentity),
                typeof(SkillSet),
                typeof(PsychologyState),
                typeof(PhysiologyState),
                typeof(SocialBonds),
                typeof(SettlerStats)
            );
            
            EntityManager.SetComponentData(entity, settler.Identity);
            EntityManager.SetComponentData(entity, settler.Skills);
            EntityManager.SetComponentData(entity, settler.Psychology);
            EntityManager.SetComponentData(entity, settler.Physiology);
            EntityManager.SetComponentData(entity, settler.Stats);
            
            // Восстановление связей
            RestoreBonds(entity, settler.Bonds);
        }
    }
}
